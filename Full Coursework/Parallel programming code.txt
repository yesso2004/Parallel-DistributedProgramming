#include <iostream>
#include <omp.h>
#include <math.h>
using namespace std;

long long ParallelSum(int A[], int Left, int Right, int GrainSize)
{
    int Num = Right - Left;

    if (Num <= GrainSize)
    {
        long long Sum = 0;
        for (int i = Left; i < Right; i++)
        {
            Sum = Sum + A[i];
        }
        return Sum;
    }

    int Mid = Left + Num / 2;

    long long X = 0;
    long long Y = 0;

#pragma omp task shared(X)
    {
        X = ParallelSum(A, Left, Mid, GrainSize);
    }

#pragma omp task shared(Y)
    {
        Y = ParallelSum(A, Mid, Right, GrainSize);
    }

#pragma omp taskwait
    return X + Y;
}

int main()
{
    const long long N = 4096;
    long long GrainSize = 256;

    int A[N];
    for (int i = 0; i < N; i++)
    {
        A[i] = 1;
    }

    long long Leaves = N / GrainSize;
    if (Leaves < 1) Leaves = 1;

   
    long long WorkT1 = N - 1;

    long long Span = log2(Leaves);

    double Parallelism = (double)WorkT1 / (double)Span;

    const long long baseOverhead = 2;

    int Threads[8] = { 1,2,4,8,16,32,64,128 };
    double T1Time = 0.0;
    long long CurrentOverhead = 0;

    cout << "Work (T1) = " << WorkT1 << "\n";
    cout << "Span = " << Span << "\n";
    cout << "Parallelism = " << Parallelism << "\n\n";

    for (int i = 0; i < 8; i++)
    {
        int CurrentThread = Threads[i];
        omp_set_num_threads(CurrentThread);

        if (CurrentThread == 1)
        {
            CurrentOverhead = 0;
        }
        else
        {
            CurrentOverhead = (Span * CurrentThread * baseOverhead);
        }

        long long BurdenedSpan = Span + CurrentOverhead;
        long long BurdenedParallelism = WorkT1 / BurdenedSpan;

        double StartTime = omp_get_wtime();
        long long Answer = 0;

#pragma omp parallel
        {
#pragma omp single
            {
                Answer = ParallelSum(A, 0, N, GrainSize);
            }
        }

        double Tp = omp_get_wtime() - StartTime;

        if (CurrentThread == 1)
        {
            T1Time = Tp;
        }

        double Speedup = T1Time / Tp;

        cout << "Threads = " << CurrentThread
            << " | Time = " << Tp
            << " | Speedup = " << Speedup
            << " | Sum = " << Answer
            << "\n";

        cout << "Burdened Span = " << BurdenedSpan
            << " | Burdened Parallelism = " << BurdenedParallelism
            << "\n\n";
    }

    return 0;
}
